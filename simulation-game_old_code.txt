const MAX_SAVES = 5;
let _saves = {};

function getSaves() {
  try { const raw = localStorage.getItem('pixelLifeSaves'); if (raw) _saves = JSON.parse(raw); } catch(e) {}
  return _saves;
}

function persistSaves() {
  try { localStorage.setItem('pixelLifeSaves', JSON.stringify(_saves)); } catch(e) {}
}

function buildSaveData() {
  return {
    coins, hp, energy, hunger, day, gameTime,
    inv: {...inv},
    houseItems: HOUSE_ITEMS.map(h => ({ id: h.id, done: h.done })),
    player: { x: player.x, y: player.y },
    season: SEASONS.current,
    weather: WEATHER.current,
    savedAt: new Date().toLocaleString(),
  };
}

function applysave(data) {
  coins = data.coins; hp = data.hp; energy = data.energy;
  hunger = data.hunger; day = data.day; gameTime = data.gameTime;
  Object.assign(inv, data.inv);
  data.houseItems.forEach(saved => {
    const item = HOUSE_ITEMS.find(h => h.id === saved.id);
    if (item) item.done = saved.done;
  });
  player.x = data.player.x; player.y = data.player.y;
  SEASONS.current = data.season; WEATHER.current = data.weather;
  updateHUD(); updateInvUI();
}

function saveGame() { openSaveModal('save'); }
function loadGame() { openSaveModal('load'); }

function openSaveModal(mode) {
  getSaves();
  const existing = document.getElementById('save-modal');
  if (existing) existing.remove();

  const overlay = document.createElement('div');
  overlay.id = 'save-modal';
  overlay.style.cssText = `position:fixed;top:0;left:0;width:100%;height:100%;background:rgba(0,0,0,0.88);z-index:999;display:flex;align-items:center;justify-content:center;font-family:'Press Start 2P',monospace;`;

  const box = document.createElement('div');
  box.style.cssText = `background:#0d1117;border:2px solid #ffd700;padding:clamp(12px,3vw,20px);width:min(92vw,480px);max-height:85vh;overflow-y:auto;-webkit-overflow-scrolling:touch;`;
  const title = document.createElement('div');
  title.textContent = mode === 'save' ? '💾 SAVE GAME' : '📂 LOAD GAME';
  title.style.cssText = `color:#ffd700;font-size:clamp(7px,2.5vw,11px);margin-bottom:clamp(10px,2vh,16px);text-align:center;`;
  box.appendChild(title);

  for (let i = 1; i <= MAX_SAVES; i++) {
    const slot = _saves[i];
    const row = document.createElement('div');
    row.style.cssText = `display:flex;align-items:center;flex-wrap:wrap;gap:clamp(4px,1.5vw,8px);margin-bottom:clamp(6px,1.5vh,10px);background:#1a2030;border:1px solid #2d3a4a;padding:clamp(6px,1.5vw,10px);`;

    const info = document.createElement('div');
    info.style.cssText = `flex:1;`;
    const slotLabel = document.createElement('div');
    slotLabel.style.cssText = `color:#6ee7b7;font-size:clamp(5px,1.8vw,7px);margin-bottom:4px;`;
    slotLabel.textContent = `SLOT ${i}`;
    const slotDetail = document.createElement('div');
    slotDetail.style.cssText = `color:#9ca3af;font-size:clamp(4px,1.4vw,6px);line-height:2;word-break:break-word;`;
    slotDetail.textContent = slot
      ? `Day ${slot.day} · 🪙${slot.coins} · ${Math.round(slot.houseItems.filter(h=>h.done).length/10*100)}% built · ${slot.savedAt}`
      : '— Empty —';
    info.appendChild(slotLabel);
    info.appendChild(slotDetail);
    row.appendChild(info);

    if (mode === 'save') {
      const btn = document.createElement('button');
      btn.textContent = slot ? 'OVERWRITE' : 'SAVE HERE';
      btn.style.cssText = `font-family:'Press Start 2P';font-size:clamp(4px,1.4vw,6px);background:${slot?'#92400e':'#166534'};color:#fff;border:none;padding:clamp(5px,1.5vw,8px) clamp(6px,2vw,12px);cursor:pointer;white-space:nowrap;touch-action:manipulation;min-height:32px;`;
      btn.onclick = () => {
        _saves[i] = buildSaveData();
        persistSaves();
        addMsg(`💾 Saved to Slot ${i}!`);
        overlay.remove();
      };
      row.appendChild(btn);
      if (slot) {
        const del = document.createElement('button');
        del.textContent = '🗑';
        del.style.cssText = `font-family:'Press Start 2P';font-size:6px;background:#7f1d1d;color:#fff;border:none;padding:6px 8px;cursor:pointer;`;
        del.onclick = () => { delete _saves[i]; persistSaves(); overlay.remove(); openSaveModal('save'); };
        row.appendChild(del);
      }
    } else {
      if (slot) {
        const btn = document.createElement('button');
        btn.textContent = 'LOAD';
        btn.style.cssText = `font-family:'Press Start 2P';font-size:clamp(4px,1.4vw,6px);background:#1e40af;color:#fff;border:none;padding:clamp(5px,1.5vw,8px) clamp(6px,2vw,12px);cursor:pointer;touch-action:manipulation;min-height:32px;`;
        btn.onclick = () => {
          applyave(_saves[i]);
          addMsg(`✅ Loaded Slot ${i}! Day ${_saves[i].day}`);
          overlay.remove();
          if (SCENE === 'menu') {
            initGame();
            applyave(_saves[i]);
            if (!loopGoing) { loopGoing = true; loop(); }
            setupMobileControls();
          }
        };
        row.appendChild(btn);
        const del = document.createElement('button');
        del.textContent = '🗑';
        del.style.cssText = `font-family:'Press Start 2P';font-size:clamp(5px,1.6vw,7px);background:#7f1d1d;color:#fff;border:none;padding:clamp(5px,1.5vw,8px) clamp(5px,1.8vw,10px);cursor:pointer;touch-action:manipulation;min-height:32px;`;
        del.onclick = () => { delete _saves[i]; persistSaves(); overlay.remove(); openSaveModal('load'); };
        row.appendChild(del);
      } else {
        const empty = document.createElement('div');
        empty.style.cssText = `color:#374151;font-size:5px;padding:6px 10px;`;
        empty.textContent = 'EMPTY';
        row.appendChild(empty);
      }
    }
    box.appendChild(row);
  }

  const close = document.createElement('button');
  close.textContent = 'CANCEL';
  close.style.cssText = `font-family:'Press Start 2P';font-size:clamp(4px,1.4vw,6px);background:#e94560;color:#fff;border:none;padding:clamp(5px,1.5vw,8px) clamp(6px,2vw,12px);cursor:pointer;margin-top:clamp(5px,1.5vw,8px);display:block;margin-left:auto;margin-right:auto;`;
  close.onclick = () => overlay.remove();
  box.appendChild(close);

  overlay.appendChild(box);
  overlay.addEventListener('mousedown', e => { if (e.target === overlay) overlay.remove(); });
  document.body.appendChild(overlay);
}

// Seasons System
const SEASONS = {
  current: 'spring',
  day: 1,
  daysPerSeason: 2, // every 2 in-game days, season changes
  list: ['spring','summer','autumn','winter'],
  leaves: [], // falling leaves for autumn
  snowflakes: [], // falling snow for winter
  petalflakes: [], // falling blossoms for spring
};

function updateSeasons() {
  const idx = Math.floor((day - 1) / SEASONS.daysPerSeason) % SEASONS.list.length;
  SEASONS.current = SEASONS.list[idx];

  // Spawn ground particles based on season
  if (SEASONS.current === 'autumn') {
    if (Math.random() < 0.3) {
      SEASONS.leaves.push({
        x: cam.x + Math.random() * W,
        y: cam.y + Math.random() * MAP_H,
        driftX: (Math.random() - 0.5) * 0.3,
        size: 4 + Math.random() * 5,
        rot: Math.random() * Math.PI * 2,
        rotSpeed: (Math.random() - 0.5) * 0.04,
        color: ['#ef4444','#f97316','#eab308','#dc2626','#b45309'][Math.floor(Math.random()*5)],
        alpha: 0.6 + Math.random() * 0.4,
        ground: true,
      });
    }
  }

  if (SEASONS.current === 'winter') {
    if (Math.random() < 0.35) {
      SEASONS.snowflakes.push({
        x: cam.x + Math.random() * MAP_W,
        y: cam.y + Math.random() * MAP_H,
        size: 2 + Math.random() * 4,
        opacity: 0.5 + Math.random() * 0.5,
        ground: true,
      });
    }
  }

  if (SEASONS.current === 'spring') {
    if (Math.random() < 0.15) {
      SEASONS.petalflakes.push({
        x: cam.x + Math.random() * MAP_W,
        y: cam.y + Math.random() * MAP_H,
        driftX: (Math.random() - 0.5) * 0.3,
        size: 3 + Math.random() * 3,
        rot: Math.random() * Math.PI * 2,
        rotSpeed: (Math.random() - 0.5) * 0.04,
        color: ['#fbcfe8','#f9a8d4','#fce7f3','#fda4af'][Math.floor(Math.random()*4)],
        alpha: 0.5 + Math.random() * 0.5,
        ground: true,
      });
    }
  }

  // Cap particle counts so it doesn't grow forever
  if (SEASONS.leaves.length     > 400) SEASONS.leaves.splice(0, 10);
  if (SEASONS.snowflakes.length > 500) SEASONS.snowflakes.splice(0, 10);
  if (SEASONS.petalflakes.length > 300) SEASONS.petalflakes.splice(0, 8);

  // Clear particles when season changes
  if (SEASONS.current !== 'autumn')  SEASONS.leaves      = [];
  if (SEASONS.current !== 'winter')  SEASONS.snowflakes  = [];
  if (SEASONS.current !== 'spring')  SEASONS.petalflakes = [];
}

function drawSeasons() {
  const season = SEASONS.current;
  const cx = cam.x, cy = cam.y;

  // Autumn leaves on ground
  for (const l of SEASONS.leaves) {
    const sx = l.x - cx, sy = l.y - cy;
    if (sx < -10 || sx > W + 10 || sy < -10 || sy > H + 10) continue;
    ctx.save();
    ctx.globalAlpha = l.alpha;
    ctx.translate(~~sx, ~~sy);
    ctx.rotate(l.rot);
    ctx.fillStyle = l.color;
    ctx.beginPath();
    ctx.ellipse(0, 0, l.size, l.size * 0.55, 0, 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
    ctx.globalAlpha = 1;
  }

  // Winter snow patches on ground
  for (const s of SEASONS.snowflakes) {
    const sx = s.x - cx, sy = s.y - cy;
    if (sx < -10 || sx > W + 10 || sy < -10 || sy > H + 10) continue;
    ctx.globalAlpha = s.opacity * 0.75;
    ctx.fillStyle = '#e0f2fe';
    ctx.beginPath();
    ctx.arc(~~sx, ~~sy, s.size, 0, Math.PI * 2);
    ctx.fill();
    ctx.globalAlpha = 1;
  }

  // Spring petals on ground
  for (const p of SEASONS.petalflakes) {
    const sx = p.x - cx, sy = p.y - cy;
    if (sx < -10 || sx > W + 10 || sy < -10 || sy > H + 10) continue;
    ctx.save();
    ctx.globalAlpha = p.alpha;
    ctx.translate(~~sx, ~~sy);
    ctx.rotate(p.rot);
    ctx.fillStyle = p.color;
    ctx.beginPath();
    ctx.ellipse(0, 0, p.size, p.size * 0.5, 0, 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
    ctx.globalAlpha = 1;
  }

  // Season label only — no tint overlay
  const icons = {
    spring: '🌸 Spring',
    summer: '☀️ Summer',
    autumn: '🍂 Autumn',
    winter: '❄️ Winter',
  };
  ctx.font = '20px Arial';
  ctx.fillStyle = 'white';
  ctx.strokeStyle = 'rgba(0,0,0,0.5)';
  ctx.lineWidth = 3;
  ctx.strokeText(icons[season], canvas.width - 120, 56);
  ctx.fillText(icons[season], canvas.width - 120, 56);
}

function updateWeather() {
  WEATHER.timer++;
  if (WEATHER.timer >= WEATHER.duration) {
    WEATHER.timer = 0;
    const types = WEATHER.types;
    const curr = types.indexOf(WEATHER.current);
    // weighted random: bias toward nearby weather states
    const roll = Math.random();
    if (roll < 0.4) WEATHER.current = types[curr]; // stay same
    else if (roll < 0.7) WEATHER.current = types[(curr + 1) % types.length];
    else if (roll < 0.9) WEATHER.current = types[(curr + 2) % types.length];
    else WEATHER.current = types[Math.floor(Math.random() * types.length)];
    WEATHER.duration = 1200 + Math.floor(Math.random() * 2400); // 20-60 sec
  }

  // Spawn raindrops
  if (WEATHER.current === 'rainy' || WEATHER.current === 'stormy') {
    const count = WEATHER.current === 'stormy' ? 6 : 3;
    for (let i = 0; i < count; i++) {
      WEATHER.raindrops.push({
        x: Math.random() * canvas.width,
        y: -10,
        speed: 6 + Math.random() * 4,
        angle: WEATHER.current === 'stormy' ? 0.3 : 0.05,
      });
    }
  }

  // Update raindrops
  WEATHER.raindrops = WEATHER.raindrops.filter(d => d.y < canvas.height + 20);
  for (const d of WEATHER.raindrops) {
    d.x += d.angle * d.speed;
    d.y += d.speed;
  }
}

function updateWanderingNPCs() {

  const isNight = tod < 0.15 || tod > 0.82;

  TOWN_NPCS.forEach(npc => {
    npc.animT++;

    if(isNight) {
      npc.state = 'gohome';
      if(!npc.homeX) {
        // Assign a house from NPC_HOUSES based on id
        const h = NPC_HOUSES[npc.id % NPC_HOUSES.length];
        npc.homeX = h.x + h.w/2;
        npc.homeY = h.y + h.h/2;
      }
      if(npc.atHome) { npc.vx = 0; npc.vy = 0; return; }
      const dx = npc.homeX - npc.x, dy = npc.homeY - npc.y;
      const dist = Math.hypot(dx, dy);
      if(dist <= 10) { npc.atHome = true; npc.vx = 0; npc.vy = 0; return; }
      const spd = 1.2;
      npc.vx = dx/dist*spd; npc.vy = dy/dist*spd;
      npc.x = Math.max(30, Math.min(MAP_W-30, npc.x + npc.vx));
      npc.y = Math.max(30, Math.min(MAP_H-30, npc.y + npc.vy));
      // Hard collision vs buildings while going home
      for(const b of BUILDINGS){
        const pad=12;
        const bx1=b.x-pad,by1=b.y-pad,bx2=b.x+b.w+pad,by2=b.y+b.h+pad;
        if(npc.x>bx1&&npc.x<bx2&&npc.y>by1&&npc.y<by2){
          const oL=npc.x-bx1,oR=bx2-npc.x,oT=npc.y-by1,oB=by2-npc.y;
          const m=Math.min(oL,oR,oT,oB);
          if(m===oL){npc.x=bx1;npc.vx=-Math.abs(npc.vx);}
          else if(m===oR){npc.x=bx2;npc.vx=Math.abs(npc.vx);}
          else if(m===oT){npc.y=by1;npc.vy=-Math.abs(npc.vy);}
          else{npc.y=by2;npc.vy=Math.abs(npc.vy);}
        }
      }
      npc.dir = Math.abs(dx) > Math.abs(dy) ? (dx>0?3:2) : (dy>0?0:1);
      if(npc.animT % 8 === 0) npc.frame = (npc.frame+1)%4;
      npc.animT++;
      return;
    }

    // Daytime — reset home state
    npc.atHome = false;

    npc.shopT--;
    if(npc.shopT <= 0 && npc.state === 'walk') {

      npc.shopWaitT = 0;

      const spd = 1.1;
      npc.vx = dx/dist*spd; npc.vy = dy/dist*spd;
      npc.x = Math.max(30, Math.min(MAP_W-30, npc.x + npc.vx));
      npc.y = Math.max(30, Math.min(MAP_H-30, npc.y + npc.vy));
      // Hard collision vs buildings while pathing
      for(const b of BUILDINGS){
        const pad=12;
        const bx1=b.x-pad,by1=b.y-pad,bx2=b.x+b.w+pad,by2=b.y+b.h+pad;
        if(npc.x>bx1&&npc.x<bx2&&npc.y>by1&&npc.y<by2){
          const oL=npc.x-bx1,oR=bx2-npc.x,oT=npc.y-by1,oB=by2-npc.y;
          const m=Math.min(oL,oR,oT,oB);
          if(m===oL){npc.x=bx1;npc.vx=-Math.abs(npc.vx);}
          else if(m===oR){npc.x=bx2;npc.vx=Math.abs(npc.vx);}
          else if(m===oT){npc.y=by1;npc.vy=-Math.abs(npc.vy);}
          else{npc.y=by2;npc.vy=Math.abs(npc.vy);}
        }
      }
      npc.dir = Math.abs(dx)>Math.abs(dy)?(dx>0?3:2):(dy>0?0:1);
      if(npc.animT%8===0) npc.frame=(npc.frame+1)%4;
      npc.animT++;
      if(dist < 22) {
        npc.state = 'shopping';
        npc.shopWaitT = 80 + Math.floor(Math.random()*80);
        npc.shopBubbleT = 0;
      }
      return;
    }

    // At the shop — stand and wait, show bubble
    if(npc.state === 'shopping') {
      npc.shopWaitT--;
      npc.shopBubbleT++;
      npc.vx = 0; npc.vy = 0;
      if(npc.shopWaitT <= 0) {
        // Done shopping, go back to walking
        npc.state = 'walk';
        npc.shopTarget = null;
        const angle = Math.random()*Math.PI*2;
        const spd2 = 0.6 + Math.random()*0.6;
        npc.vx = Math.cos(angle)*spd2;
        npc.vy = Math.sin(angle)*spd2;
      }
      return;
    }

    if(npc.state === 'idle') {
      npc.idleT--;
      if(npc.idleT <= 0) { npc.state = 'walk'; }
      return;
    }

    // Move
    npc.x += npc.vx;
    npc.y += npc.vy;

    // Bounce off MAP bounds with margin
    const margin = 30;
    if(npc.x < margin) { npc.vx = Math.abs(npc.vx); npc.x = margin; }
    if(npc.x > MAP_W - margin) { npc.vx = -Math.abs(npc.vx); npc.x = MAP_W - margin; }
    if(npc.y < margin) { npc.vy = Math.abs(npc.vy); npc.y = margin; }
    if(npc.y > MAP_H - margin) { npc.vy = -Math.abs(npc.vy); npc.y = MAP_H - margin; }

    // Avoid buildings (soft repulsion)
    // Hard collision vs buildings — push NPC outside the box
    for(const b of BUILDINGS) {
      const pad = 14;
      const bx1 = b.x - pad, by1 = b.y - pad;
      const bx2 = b.x + b.w + pad, by2 = b.y + b.h + pad;
      if(npc.x > bx1 && npc.x < bx2 && npc.y > by1 && npc.y < by2) {
        // Find shallowest exit axis and push out
        const overL = npc.x - bx1, overR = bx2 - npc.x;
        const overT = npc.y - by1, overB = by2 - npc.y;
        const minOver = Math.min(overL, overR, overT, overB);
        if(minOver === overL) { npc.x = bx1; npc.vx = Math.min(npc.vx, -0.5); }
        else if(minOver === overR) { npc.x = bx2; npc.vx = Math.max(npc.vx,  0.5); }
        else if(minOver === overT) { npc.y = by1; npc.vy = Math.min(npc.vy, -0.5); }
        else                       { npc.y = by2; npc.vy = Math.max(npc.vy,  0.5); }
      }
    }

    // Avoid tennis and basketball courts
    for(const court of [TENNIS_COURT, BBALL_COURT, PLAYGROUND, BASEBALL_FIELD]) {
      const cCX = court.x + court.w/2, cCY = court.y + court.h/2;
      const dx = npc.x - cCX, dy = npc.y - cCY;
      if(Math.abs(dx) < court.w/2+25 && Math.abs(dy) < court.h/2+25) {
        const dist = Math.hypot(dx, dy) || 1;
        npc.vx += dx/dist * 0.4;
        npc.vy += dy/dist * 0.4;
      }
    }

    // Clamp speed
    const spd = Math.hypot(npc.vx, npc.vy);
    const maxSpd = 1.5;
    if(spd > maxSpd) { npc.vx = npc.vx/spd*maxSpd; npc.vy = npc.vy/spd*maxSpd; }

    // Direction for sprite
    if(Math.abs(npc.vx) > Math.abs(npc.vy)) {
      npc.dir = npc.vx > 0 ? 3 : 2;
    } else {
      npc.dir = npc.vy > 0 ? 0 : 1;
    }

    // Animate walk
    if(npc.animT % 8 === 0) npc.frame = (npc.frame+1)%4;

    // Periodically change direction or idle
    npc.changeDirT--;
    if(npc.changeDirT <= 0) {
      npc.changeDirT = 80 + Math.floor(Math.random()*200);
      if(Math.random() < 0.25) {
        npc.state = 'idle';
        npc.idleT = 40 + Math.floor(Math.random()*80);
        npc.vx = 0; npc.vy = 0;
      } else {
        // Drift toward preferred zone center
        const z = npc.zone;
        const zCX = (z.x1+z.x2)/2, zCY = (z.y1+z.y2)/2;
        const dx = zCX - npc.x, dy = zCY - npc.y;
        const dist = Math.hypot(dx,dy) || 1;
        const baseSpd = 0.6 + Math.random()*0.8;
        // Mix of random direction + pull toward zone
        const randA = Math.random()*Math.PI*2;
        const zonePull = Math.min(1, dist/400); // stronger pull when far from zone
        npc.vx = (Math.cos(randA)*(1-zonePull) + dx/dist*zonePull) * baseSpd;
        npc.vy = (Math.sin(randA)*(1-zonePull) + dy/dist*zonePull) * baseSpd;
      }
    }
  });
}

function updateCourtNPCs() {
  globalT++;
  const tc = TENNIS_COURT;
  const bc = BBALL_COURT;

  // ── TENNIS NPC ANIMATION ──
  if(tennisNPCs.initialized) {
    const tn = tennisNPCs;
    const tp1 = tn.p1, tp2 = tn.p2;
    tn.phaseT++;

    // Move p1 (bottom) tracking ball X
    const p1targetX = Math.max(tc.x+20, Math.min(tc.x+tc.w-20, tn.bX + (Math.random()-0.5)*10));
    tp1.x += (p1targetX - tp1.x) * 0.06;
    tp1.y = tc.y + tc.h*0.78 + Math.sin(globalT*0.03)*8;
    tp1.animT++; if(tp1.animT%8===0) tp1.frame=(tp1.frame+1)%4;

    // Move p2 (top) tracking ball X  
    const p2targetX = Math.max(tc.x+20, Math.min(tc.x+tc.w-20, tn.bX + (Math.random()-0.5)*10));
    tp2.x += (p2targetX - tp2.x) * 0.06;
    tp2.y = tc.y + tc.h*0.22 + Math.sin(globalT*0.03+1.5)*8;
    tp2.animT++; if(tp2.animT%8===0) tp2.frame=(tp2.frame+1)%4;

    // Ball bounce in court
    if(tn.ballActive) {
      tn.trail.push({x:tn.bX, y:tn.bY});
      if(tn.trail.length > 12) tn.trail.shift();

      tn.bX += tn.bVX;
      tn.bY += tn.bVY;

      // Bounce off left/right walls
      if(tn.bX < tc.x+8) { tn.bVX = Math.abs(tn.bVX); tn.bX = tc.x+8; }
      if(tn.bX > tc.x+tc.w-8) { tn.bVX = -Math.abs(tn.bVX); tn.bX = tc.x+tc.w-8; }

      // When ball reaches p1 side (bottom), p1 hits it back
      if(tn.bY > tc.y+tc.h*0.72 && tn.bVY > 0) {
        // Hit back toward p2
        const targetX = tc.x + tc.w*(0.2+Math.random()*0.6);
        const dx = targetX - tn.bX;
        const spd = 1.4 + Math.random()*0.6;
        tn.bVX = dx/Math.abs(dx)*Math.abs(dx/100)*spd;
        tn.bVY = -(1.0+Math.random()*0.6);
        tp1.swinging = true; tp1.swingT = 0;
        tn.rallyCount++;
      }

      // When ball reaches p2 side (top), p2 hits it back
      if(tn.bY < tc.y+tc.h*0.28 && tn.bVY < 0) {
        const targetX = tc.x + tc.w*(0.2+Math.random()*0.6);
        const dx = targetX - tn.bX;
        const spd = 1.4 + Math.random()*0.6;
        tn.bVX = dx/Math.abs(dx)*Math.abs(dx/100)*spd;
        tn.bVY = (1.0+Math.random()*0.6);
        tp2.swinging = true; tp2.swingT = 0;
        tn.rallyCount++;
      }

      // Keep ball in vertical range
      if(tn.bY < tc.y+10) { tn.bVY = Math.abs(tn.bVY); tn.bY = tc.y+10; }
      if(tn.bY > tc.y+tc.h-10) { tn.bVY = -Math.abs(tn.bVY); tn.bY = tc.y+tc.h-10; }

      // Add some gravity-like deceleration
      tn.bVX *= 0.998;
    }

    // Swing animation countdown
    if(tp1.swinging) { tp1.swingT++; if(tp1.swingT>16) tp1.swinging=false; }
    if(tp2.swinging) { tp2.swingT++; if(tp2.swingT>16) tp2.swinging=false; }
  }

  // ── BASKETBALL NPC ANIMATION ──
  if(bballNPCs.initialized) {
    const bn = bballNPCs;
    const bp1 = bn.p1, bp2 = bn.p2;
    bn.phaseT++;
    bn.dribbleT += 0.2;

    bp1.animT++; if(bp1.animT%8===0) bp1.frame=(bp1.frame+1)%4;
    bp2.animT++; if(bp2.animT%8===0) bp2.frame=(bp2.frame+1)%4;

      const hoopX = bc.x + bc.w - 40;
      const hoopY = bc.y + bc.h/2;
      const dx = hoopX - bp1.x, dy = hoopY - bp1.y;
      const dist = Math.hypot(dx,dy);
      if(dist > 8) {
        const s = Math.min(dist, 1.0);
        bp1.x += dx/dist*s;
        bp1.y += dy/dist*s;
        bp1.dir = dx > 0 ? 3 : 2;
      }
      // p2 defends — positions between p1 and hoop
      const defX = bp1.x*0.5 + hoopX*0.5;
      const defY = bp1.y*0.5 + hoopY*0.5;
      const dx2 = defX - bp2.x, dy2 = defY - bp2.y;
      const d2 = Math.hypot(dx2,dy2);
      if(d2 > 5) {
        bp2.x += dx2/d2 * 0.8;
        bp2.y += dy2/d2 * 0.8;
        bp2.dir = dx2 > 0 ? 3 : 2;
      }

      // Ball dribbles with p1
      bp1.hasBall = true;
      bn.bX = bp1.x + (bp1.dir===3?10:-10);
      bn.bY = bp1.y - 12 + Math.abs(Math.sin(bn.dribbleT))*14;

      // When close to hoop, shoot
      if(dist < 90 || bn.phaseT > 200) {
        bn.ballPhase = 'shooting';
        bn.phaseT = 0;
        bn.shotT = 0;
        bn.shotDur = 38;
        bn.shotSX = bp1.x; bn.shotSY = bp1.y;
        bn.shotTX = hoopX; bn.shotTY = hoopY;
        bp1.hasBall = false;
      }
    }
    else if(bn.ballPhase === 'shooting') {
      // Animate ball arc toward hoop
      bn.shotT++;
      const t = bn.shotT / bn.shotDur;
      bn.bX = bn.shotSX + (bn.shotTX - bn.shotSX)*t;
      bn.bY = bn.shotSY + (bn.shotTY - bn.shotSY)*t - Math.sin(t*Math.PI)*80;

      if(bn.shotT >= bn.shotDur) {
        // Score or miss — random
        const scored = Math.random() < 0.65;
        bn.ballPhase = 'reset';
        bn.phaseT = 0;
      }
    }
    else if(bn.ballPhase === 'reset') {
      // Short pause, then p2 gets the ball and dribbles back
      bn.phaseT++;
      if(bn.phaseT > 60) {
        // Swap roles: p2 now has ball, dribbles to left hoop
        bn.ballPhase = 'dribble2';
        bn.phaseT = 0;
        bp2.hasBall = true;
        bp1.hasBall = false;
        // Reset positions
        bp2.x = bc.x + bc.w*0.7; bp2.y = bc.y + bc.h/2;
        bp1.x = bc.x + bc.w*0.3; bp1.y = bc.y + bc.h/2;
      }
    }
    else if(bn.ballPhase === 'dribble2') {
      // p2 dribbles toward left hoop
      const hoopX = bc.x + 40;
      const hoopY = bc.y + bc.h/2;
      const dx = hoopX - bp2.x, dy = hoopY - bp2.y;
      const dist = Math.hypot(dx,dy);
      if(dist > 8) {
        const s = Math.min(dist, 1.0);
        bp2.x += dx/dist*s;
        bp2.y += dy/dist*s;
        bp2.dir = dx > 0 ? 3 : 2;
      }
      // p1 defends
      const defX = bp2.x*0.5 + hoopX*0.5;
      const defY = bp2.y*0.5 + hoopY*0.5;
      const dx2 = defX - bp1.x, dy2 = defY - bp1.y;
      const d2 = Math.hypot(dx2,dy2);
      if(d2 > 5) {
        bp1.x += dx2/d2 * 0.8;
        bp1.y += dy2/d2 * 0.8;
        bp1.dir = dx2 > 0 ? 3 : 2;
      }

      bp2.hasBall = true;
      bn.bX = bp2.x + (bp2.dir===3?10:-10);
      bn.bY = bp2.y - 12 + Math.abs(Math.sin(bn.dribbleT))*14;

      if(dist < 90 || bn.phaseT > 200) {
        bn.ballPhase = 'shooting2';
        bn.phaseT = 0;
        bn.shotT = 0;
        bn.shotDur = 38;
        bn.shotSX = bp2.x; bn.shotSY = bp2.y;
        bn.shotTX = hoopX; bn.shotTY = hoopY;
        bp2.hasBall = false;
      }
    }
    else if(bn.ballPhase === 'shooting2') {
      bn.shotT++;
      const t = bn.shotT / bn.shotDur;
      bn.bX = bn.shotSX + (bn.shotTX - bn.shotSX)*t;
      bn.bY = bn.shotSY + (bn.shotTY - bn.shotSY)*t - Math.sin(t*Math.PI)*80;
      if(bn.shotT >= bn.shotDur) {
        const scored = Math.random() < 0.65;
        bn.ballPhase = 'reset2';
        bn.phaseT = 0;
      }
    }
    else if(bn.ballPhase === 'reset2') {
      bn.phaseT++;
      if(bn.phaseT > 60) {
        bn.ballPhase = 'dribble';
        bn.phaseT = 0;
        bp1.hasBall = true;
        bp2.hasBall = false;
        bp1.x = bc.x + bc.w*0.3; bp1.y = bc.y + bc.h/2;
        bp2.x = bc.x + bc.w*0.7; bp2.y = bc.y + bc.h/2;
        bn.bX = bp1.x; bn.bY = bp1.y-14;
      }
    }

    // Clamp NPC positions inside court
    for(const p of [bp1,bp2]) {
      p.x = Math.max(bc.x+14, Math.min(bc.x+bc.w-14, p.x));
      p.y = Math.max(bc.y+14, Math.min(bc.y+bc.h-14, p.y));
    }
  }

  // ── BASEBALL NPC BACKGROUND ANIMATION ──
  if(baseballNPCs.initialized) {
    const bn = baseballNPCs;
    const bf = BASEBALL_FIELD;
    const homeX = bf.x + bf.w*0.5, homeY = bf.y + bf.h*0.85;
    const firstX = bf.x + bf.w*0.82, firstY = bf.y + bf.h*0.55;
    const secondX = bf.x + bf.w*0.5, secondY = bf.y + bf.h*0.22;
    const pitcherX = bf.x + bf.w*0.5, pitcherY = bf.y + bf.h*0.38;
    bn.phaseT++;

    // Animate characters
    bn.pitcher.animT++; if(bn.pitcher.animT%10===0) bn.pitcher.frame=(bn.pitcher.frame+1)%4;
    bn.batter.animT++;  if(bn.batter.animT%8===0)  bn.batter.frame=(bn.batter.frame+1)%4;
    bn.runner.animT++;  if(bn.runner.animT%6===0)  bn.runner.frame=(bn.runner.frame+1)%4;

    if(bn.phase === 'windup') {
      bn.pitcher.x = pitcherX; bn.pitcher.y = pitcherY; bn.pitcher.dir = 0;
      bn.batter.x = homeX - 18; bn.batter.y = homeY; bn.batter.dir = 2; bn.batter.swinging = false;
      bn.runner.running = false; bn.runner.x = homeX; bn.runner.y = homeY+8;
      if(bn.phaseT > 80) {
        bn.phase = 'pitch'; bn.phaseT = 0;
        bn.bX = pitcherX; bn.bY = pitcherY;
        bn.bVX = (homeX - pitcherX) / 40;
        bn.bVY = (homeY - pitcherY) / 40;
        bn.ballActive = true; bn.inFlight = true;
        bn.trail = [];
      }
    } else if(bn.phase === 'pitch') {
      bn.trail.push({x:bn.bX, y:bn.bY}); if(bn.trail.length>10) bn.trail.shift();
      bn.bX += bn.bVX; bn.bY += bn.bVY;
      const dist = Math.hypot(bn.bX - homeX, bn.bY - homeY);
      if(dist < 12) {
        // Decide: hit or miss
        if(Math.random() < 0.7) {
          bn.phase = 'hit'; bn.phaseT = 0;
          bn.batter.swinging = true; bn.batter.swingT = 0;
          const ang = -0.5 + Math.random()*1.0;
          bn.bVX = Math.cos(ang) * 4.5;
          bn.bVY = -3.2;
        } else {
          bn.phase = 'miss'; bn.phaseT = 0;
          bn.batter.swinging = true; bn.batter.swingT = 0;
          bn.ballActive = false;
        }
      }
    } else if(bn.phase === 'hit') {
      bn.batter.swingT++;
      if(bn.batter.swingT > 12) bn.batter.swinging = false;
      bn.trail.push({x:bn.bX, y:bn.bY}); if(bn.trail.length>10) bn.trail.shift();
      bn.bX += bn.bVX; bn.bVY += 0.12; bn.bY += bn.bVY;
      // Bounce off field edges
      if(bn.bX < bf.x+10) { bn.bVX = Math.abs(bn.bVX)*0.7; bn.bX = bf.x+10; }
      if(bn.bX > bf.x+bf.w-10) { bn.bVX = -Math.abs(bn.bVX)*0.7; bn.bX = bf.x+bf.w-10; }
      if(bn.bY < bf.y+10) { bn.bVY = Math.abs(bn.bVY)*0.5; bn.bY = bf.y+10; }
      if(bn.bY > homeY) { bn.bVY = -Math.abs(bn.bVY)*0.4; bn.bY = homeY; if(Math.abs(bn.bVY)<0.5) bn.ballActive=false; }
      // Runner starts running toward first
      bn.runner.running = true;
      const rdx = firstX - bn.runner.x, rdy = firstY - bn.runner.y;
      const rd = Math.hypot(rdx, rdy);
      if(rd > 6) { bn.runner.x += rdx/rd*1.5; bn.runner.y += rdy/rd*1.5; bn.runner.dir = rdx>0?3:2; }
      if(rd < 10 && bn.phaseT > 80) { bn.phase = 'windup'; bn.phaseT = 0; bn.ballActive=false; }
    } else if(bn.phase === 'miss') {
      bn.phaseT++;
      if(bn.phaseT > 60) { bn.phase = 'windup'; bn.phaseT = 0; }
    }
  }
}

  const armL=(f===0||f===2)?-2*s:1*s, armR=(f===0||f===2)?2*s:-1*s;

const POINT_LABELS=['0','15','30','40'];
function tennisPointLabel(p){ return POINT_LABELS[Math.min(p,3)]; }

function tennisAwardPoint(winner){
  const ts=tennisState;
  ts.ballActive=false;
  ts.phase='pointover';
  ts.nextPointT=110;

  if(ts.inTiebreak){
    if(winner==='player') ts.tbPPoints++; else ts.tbAiPoints++;
    const pp=ts.tbPPoints, ap=ts.tbAiPoints;
    if((pp>=7||ap>=7)&&Math.abs(pp-ap)>=2){
      tennisAwardGame(winner,true); return;
    }
    ts.pointMsg=(winner==='player'?'YOUR POINT  ':'CPU POINT  ')+pp+'-'+ap+' TB';
    ts.pointMsgT=90; ts.serveFault=0;
    ts.server=((pp+ap)%2===1)?(ts.server==='player'?'ai':'player'):ts.server;
    return;
  }

  if(winner==='player') ts.pPoints++; else ts.aiPoints++;
  const pp=ts.pPoints, ap=ts.aiPoints;

  if(pp>=3&&ap>=3){
    if(pp===ap){ ts.deuce=true; ts.advantage=''; ts.pointMsg='DEUCE'; }
    else if(ts.deuce||ts.advantage){
      if(ts.advantage===winner){ tennisAwardGame(winner); return; }
      else if(ts.advantage){ ts.advantage=''; ts.deuce=true; ts.pointMsg='DEUCE'; }
      else { ts.advantage=winner; ts.deuce=false; ts.pointMsg='ADV '+(winner==='player'?'YOU':'CPU'); }
    } else { ts.advantage=winner; ts.deuce=false; ts.pointMsg='ADV '+(winner==='player'?'YOU':'CPU'); }
    ts.pointMsgT=90; ts.serveFault=0;
  } else if(pp>=4){ tennisAwardGame('player'); return; }
  else if(ap>=4){ tennisAwardGame('ai'); return; }
  else {
    ts.pointMsg=(winner==='player'?'YOUR POINT!  ':'CPU POINT!  ')+tennisPointLabel(pp)+' - '+tennisPointLabel(ap);
    ts.pointMsgT=90; ts.serveFault=0;
  }
}

function tennisAwardGame(winner, fromTB=false){
  const ts=tennisState;
  if(winner==='player') ts.pGames++; else ts.aiGames++;
  ts.pPoints=0; ts.aiPoints=0; ts.deuce=false; ts.advantage='';
  ts.server=(ts.server==='player'?'ai':'player');
  ts.serveFault=0;
  const pg=ts.pGames, ag=ts.aiGames;
  ts.pointMsg='GAME '+(winner==='player'?'YOU':'CPU')+'!   '+pg+' - '+ag;
  ts.pointMsgT=120;
  if(pg===6&&ag===6&&!fromTB){ ts.inTiebreak=true; ts.tbPPoints=0; ts.tbAiPoints=0; ts.pointMsg='TIEBREAK! 6-6'; ts.pointMsgT=130; return; }
  if((pg>=6||ag>=6)&&Math.abs(pg-ag)>=2){ tennisAwardSet(winner); return; }
  if(pg===7||ag===7){ tennisAwardSet(winner); }
}

function tennisAwardSet(winner){
  const ts=tennisState;
  if(winner==='player') ts.pSets++; else ts.aiSets++;
  ts.pGames=0; ts.aiGames=0;
  ts.pointMsg='SET '+(winner==='player'?'YOU':'CPU')+'!   '+ts.pSets+' - '+ts.aiSets+' sets';
  ts.pointMsgT=160;
  if(ts.pSets>=2){ ts.matchResult='won'; coins+=50; addMsg('🏆 Match Won! +50🪙'); updateHUD(); }
  else if(ts.aiSets>=2){ ts.matchResult='lost'; coins+=10; addMsg('😔 Match Lost. +10🪙'); updateHUD(); }
}

function tServe(hitter){
  const ts=tennisState;
  const sx= hitter==='player'?ts.pX:ts.aiX;
  const sy= hitter==='player'?ts.pY:ts.aiY;
  const targetX=ts.cX+ts.cW*(0.2+Math.random()*0.6);
  const targetY=hitter==='player'
    ? ts.netY-(ts.netY-ts.cY)*(0.3+Math.random()*0.5)
    : ts.netY+(ts.cY+ts.cH-ts.netY)*(0.3+Math.random()*0.5);
  const dx=targetX-sx, dy=targetY-sy;
  const dist=Math.hypot(dx,dy);
  const spd=4.5+Math.random()*1.5;
  tFireBallVel(sx,sy, dx/dist*spd, dy/dist*spd, hitter);
}

function tRally(hitter){
  const ts=tennisState;
  const sx=hitter==='player'?ts.pX:ts.aiX;
  const sy=hitter==='player'?ts.pY:ts.aiY;
  const targetX=ts.cX+ts.cW*(0.15+Math.random()*0.7);
  const targetY=hitter==='player'
    ? ts.cY+ts.cH*(0.05+Math.random()*0.35)
    : ts.cY+ts.cH*(0.6+Math.random()*0.35);
  const dx=targetX-sx, dy=targetY-sy;
  const dist=Math.hypot(dx,dy);
  const spd=4.0+Math.random()*2.0;
  tFireBallVel(sx,sy, dx/dist*spd, dy/dist*spd, hitter);
}

function updateTennis(){
  const ts=tennisState;
  if(ts.matchResult) return;
  if(ts.pointMsgT>0) ts.pointMsgT--;
  if(ts.hitFlash>0) ts.hitFlash--;

  if(ts.phase==='pointover'){
    ts.nextPointT--;
    if(ts.nextPointT<=0){
      ts.phase='pre';
      ts.pX=ts.cX+ts.cW/2; ts.pY=ts.cY+ts.cH-30;
      ts.aiX=ts.cX+ts.cW/2; ts.aiY=ts.cY+30;
      ts.ballActive=false;
      ts.pointMsg=(ts.server==='player'?'YOU SERVE':'CPU SERVES')+' — PRESS SPACE';
      ts.pointMsgT=999;
    }
    return;
  }

  let pmx=0, pmy=0;
  if(keys['ArrowLeft']||keys['a']||keys['A'])  pmx=-ts.pSpd;
  if(keys['ArrowRight']||keys['d']||keys['D']) pmx= ts.pSpd;
  if(keys['ArrowUp']||keys['w']||keys['W'])    pmy=-ts.pSpd;
  if(keys['ArrowDown']||keys['s']||keys['S'])  pmy= ts.pSpd;
  if(pmx&&pmy){pmx*=0.707;pmy*=0.707;}
  ts.pX=Math.max(ts.cX+8, Math.min(ts.cX+ts.cW-8, ts.pX+pmx));
  ts.pY=Math.max(ts.netY+8, Math.min(ts.cY+ts.cH-8, ts.pY+pmy));
  if(pmx||pmy){ ts.pAnimT++; if(ts.pAnimT%6===0) ts.pFrame=(ts.pFrame+1)%4; }
  if(ts.pHitCD>0) ts.pHitCD--;
  if(ts.pSwing){ ts.pSwingT++; if(ts.pSwingT>14){ts.pSwing=false;ts.pSwingT=0;} }

  if(ts.ballActive){
    const aiTargetX=ts.bX;
    const adx=aiTargetX-ts.aiX;
    ts.aiX+=Math.sign(adx)*Math.min(Math.abs(adx), ts.aiSpd);
  }
  ts.aiX=Math.max(ts.cX+8, Math.min(ts.cX+ts.cW-8, ts.aiX));
  ts.aiY=Math.max(ts.cY+8, Math.min(ts.netY-8, ts.aiY));
  if(ts.aiHitCD>0) ts.aiHitCD--;
  if(ts.aiSwing){ ts.aiSwingT++; if(ts.aiSwingT>14){ts.aiSwing=false;ts.aiSwingT=0;} }
  ts.aiAnimT++; if(ts.aiAnimT%6===0) ts.aiFrame=(ts.aiAnimT/6)%4;

  if(ts.phase==='pre'){
    if(ts.server==='ai'){
      ts.aiServeDelay=(ts.aiServeDelay||0)+1;
      if(ts.aiServeDelay>80){
        ts.aiServeDelay=0;
        ts.phase='rally';
        ts.pointMsg=''; ts.pointMsgT=0;
        ts.aiSwing=true; ts.aiSwingT=0;
        tServe('ai');
        sndTennisHit();
      }
    }
    if(ts.swingQueued||keys[' ']){
      ts.swingQueued=false;
      if(ts.server==='player'){
        ts.aiServeDelay=0;
        ts.phase='rally';
        ts.pointMsg=''; ts.pointMsgT=0;
        ts.pSwing=true; ts.pSwingT=0;
        tServe('player');
        sndTennisHit();
      }
    }
    return;
  }

  if(ts.phase==='rally'&&ts.ballActive){
    ts.trail.push({x:ts.bX,y:ts.bY});
    if(ts.trail.length>16) ts.trail.shift();

    ts.bX+=ts.bVX;
    ts.bY+=ts.bVY;

    if(ts.bX < ts.cX || ts.bX > ts.cX + ts.cW){
      const hitterOut = ts.lastHitter;
      ts.pointMsg = hitterOut==='player' ? 'YOUR shot OUT! CPU point' : 'CPU shot OUT! YOUR point';
      ts.pointMsgT = 90;
      sndTennisScore(hitterOut==='ai');
      tennisAwardPoint(hitterOut==='player' ? 'ai' : 'player'); return;
    }

    if(ts.bY>ts.netY && ts.lastHitter!=='player'){
      if(ts.bY>=ts.cY+ts.cH){
        ts.pointMsg='OUT! CPU scores'; ts.pointMsgT=90;
        sndTennisScore(false); tennisAwardPoint('ai'); return;
      }
      const distToPlayer=Math.hypot(ts.bX-ts.pX, ts.bY-ts.pY);
      if(distToPlayer<28&&(ts.swingQueued||keys[' '])&&ts.pHitCD===0){
        ts.swingQueued=false;
        ts.pSwing=true; ts.pSwingT=0; ts.pHitCD=20; ts.hitFlash=8;
        sndTennisHit();
        tRally('player');
        return;
      }
      if(ts.bY>ts.cY+ts.cH+10){
        ts.pointMsg='MISSED! CPU scores'; ts.pointMsgT=90;
        sndTennisScore(false); tennisAwardPoint('ai'); return;
      }
    }

    if(ts.bY<ts.netY && ts.lastHitter!=='ai'){
      if(ts.bY<=ts.cY){
        ts.pointMsg='OUT! YOUR point'; ts.pointMsgT=90;
        sndTennisScore(true); tennisAwardPoint('player'); return;
      }
      const distToAi=Math.hypot(ts.bX-ts.aiX, ts.bY-ts.aiY);
      if(distToAi<34&&ts.aiHitCD===0){
        ts.aiSwing=true; ts.aiSwingT=0; ts.aiHitCD=22;
        sndTennisHit();
        tRally('ai');
        return;
      }
      if(ts.bY<ts.cY-10){
        ts.pointMsg='CPU missed! YOUR point'; ts.pointMsgT=90;
        sndTennisScore(true); tennisAwardPoint('player'); return;
      }
    }
  }
}

function updateBball(){
  const bs=bballState;
  if(bs.result) return;
  bs.gameTime--;
  if(bs.playerScore>=21){ bs.result='won'; coins+=40; addMsg('🏆 Basketball Won! +40🪙'); updateHUD(); return; }
  if(bs.aiScore>=21){ bs.result='lost'; coins+=5; addMsg('😔 Basketball Lost. +5🪙'); updateHUD(); return; }
  if(bs.flashScore>0) bs.flashScore--;
  if(bs.swishFlash>0) bs.swishFlash--;
  if(bs.blockCD>0) bs.blockCD--;
  if(bs.stealCD>0) bs.stealCD--;
  if(bs.blockAnim>0) bs.blockAnim--;
  if(bs.stealAnim>0) bs.stealAnim--;

  if(bs.shotMeterActive){
    bs.shotMeterPower+=bs.shotMeterDir*2.2;
    if(bs.shotMeterPower>=100){bs.shotMeterPower=100;bs.shotMeterDir=-1;}
    if(bs.shotMeterPower<=0){bs.shotMeterPower=0;bs.shotMeterDir=1;}
  }

  const spd=isMobile()?3.2:2.6;
  let mx=0,my=0;
  if(!bs.shooting){
    if(keys['ArrowLeft']||keys['a']||keys['A']){mx=-spd;bs.playerDir=2;}
    if(keys['ArrowRight']||keys['d']||keys['D']){mx=spd;bs.playerDir=3;}
    if(keys['ArrowUp']||keys['w']||keys['W']){my=-spd;}
    if(keys['ArrowDown']||keys['s']||keys['S']){my=spd;}
    if(mx&&my){mx*=0.707;my*=0.707;}
  }
  bs.playerX=Math.max(bs.courtX+14,Math.min(bs.courtX+bs.courtW-14,bs.playerX+mx));
  bs.playerY=Math.max(bs.courtY+14,Math.min(bs.courtY+bs.courtH-14,bs.playerY+my));
  if(mx||my){bs.playerAnimT++;if(bs.playerAnimT%7===0)bs.playerFrame=(bs.playerFrame+1)%4;}

  if(bs.playerHasBall&&!bs.shooting){
    bs.dribbleT+=0.24;
    bs.ballX=bs.playerX+(bs.playerDir===3?10:-10);
    bs.ballY=bs.playerY-12+Math.abs(Math.sin(bs.dribbleT))*12;
    if(Math.floor(bs.dribbleT*3)%6===0) sndDribble();
  }

  if((keys['q']||keys['Q'])&&bs.stealCD===0&&bs.aiHasBall&&!bs.shooting){
    const d=Math.hypot(bs.playerX-bs.aiPosX,bs.playerY-bs.aiPosY);
    if(d<48){
      bs.stealCD=90; bs.stealAnim=30;
      if(Math.random()<0.52){
        bs.aiHasBall=false; bs.playerHasBall=true;
        addFloat2(bs.playerX,bs.playerY-28,'STEAL! 💪','#4ade80');
        sndTennisHit();
      } else {
        addFloat2(bs.playerX,bs.playerY-28,'STEAL FAILED','#94a3b8');
      }
    } else { addFloat2(bs.playerX,bs.playerY-28,'TOO FAR!','#e94560'); }
  }

  if((keys['f']||keys['F'])&&bs.blockCD===0&&!bs.shooting){
    bs.blockCD=80; bs.blockAnim=25;
    if(bs.aiShooting){
      const d=Math.hypot(bs.playerX-bs.aiPosX,bs.playerY-bs.aiPosY);
      if(d<55){
        bs.aiShooting=false; bs.aiHasBall=false; bs.playerHasBall=true;
        bs.aiShotArcT=999;
        addFloat2(bs.playerX,bs.playerY-30,'BLOCKED! 🙌','#fde047');
        sndTennisHit();
      } else { addFloat2(bs.playerX,bs.playerY-28,'OUT OF RANGE','#94a3b8'); }
    } else { addFloat2(bs.playerX,bs.playerY-28,'BLOCK READY!','#fde047'); }
  }

  if(bs.shooting){
    bs.shotArcT++;
    const dur=44;
    const t=bs.shotArcT/dur;
    bs.ballX=bs.shotStartX+(bs.shotTargetX-bs.shotStartX)*t;
    bs.ballY=bs.shotStartY+(bs.shotTargetY-bs.shotStartY)*t-Math.sin(t*Math.PI)*100;
    if(bs.shotArcT>=dur){
      bs.shooting=false;
      const distToHoop=Math.hypot(bs.shotStartX-bs.hoopRX,bs.shotStartY-bs.hoopRY);
      const diff=Math.abs(bs.shotMeterPower-72);
      const base=distToHoop<120?0.80:distToHoop<240?0.64:0.46;
      const bonus=diff<8?0.15:diff<18?0.05:-0.12;
      const acc=Math.max(0.1,Math.min(0.96,base+bonus));
      if(Math.random()<acc){
        bs.playerScore++;
        bs.flashScore=70; bs.swishFlash=45;
        const is3=distToHoop>240;
        if(is3) coins+=3; else coins+=1;
        addFloat2(bs.courtX+bs.courtW/2,bs.courtY-12,is3?'3 PTR! 🔥':'SCORE! 🏀','#4ade80');
        sndScore(); updateHUD();
        bs.aiHasBall=true; bs.playerHasBall=false;
        bs.playerX=bs.courtX+bs.courtW*0.15; bs.playerY=bs.courtY+bs.courtH/2;
        bs.aiPosX=bs.courtX+bs.courtW*0.85; bs.aiPosY=bs.courtY+bs.courtH/2;
      } else {
        addFloat2(bs.ballX,bs.ballY-10,'MISS!','#94a3b8');
        sndMiss();
        if(Math.random()<0.5){ bs.playerHasBall=true; }
        else { bs.aiHasBall=true; }
      }
      bs.shotMeterActive=false; bs.shotMeterPower=0;
    }
  }

  bs.aiAnimT++; if(bs.aiAnimT%7===0) bs.aiFrame=(bs.aiFrame+1)%4;
  bs.aiMoveT++;

  if(bs.aiShooting){
    bs.aiShotArcT++;
    const dur=44;
    const t=bs.aiShotArcT/dur;
    bs.aiBallX=bs.aiShotStartX+(bs.aiShotTargetX-bs.aiShotStartX)*t;
    bs.aiBallY=bs.aiShotStartY+(bs.aiShotTargetY-bs.aiShotStartY)*t-Math.sin(t*Math.PI)*100;
    if(bs.aiShotArcT>=dur){
      bs.aiShooting=false;
      const acc=0.62+Math.random()*0.22;
      if(Math.random()<acc){
        bs.aiScore++;
        bs.flashScore=70;

        sndScore();

        sndMiss();

        else { bs.aiHasBall=true; }
      }
    }
  } else if(bs.aiHasBall){
    const targetX=bs.hoopLX+70;
    const targetY=bs.courtY+bs.courtH/2+(Math.sin(bs.aiMoveT*0.04)*40);
    const dx=targetX-bs.aiPosX, dy=targetY-bs.aiPosY;
    const d=Math.hypot(dx,dy);
    if(d>4){const s=Math.min(d,1.6);bs.aiPosX+=dx/d*s;bs.aiPosY+=dy/d*s;}
    bs.aiPosX=Math.max(bs.courtX+14,Math.min(bs.courtX+bs.courtW-14,bs.aiPosX));
    bs.aiPosY=Math.max(bs.courtY+14,Math.min(bs.courtY+bs.courtH-14,bs.aiPosY));

    if(bs.aiShotTimer<=0||(distToHoop<130&&Math.random()<0.025)){
      bs.aiHasBall=false;
      bs.aiShooting=true; bs.aiShotArcT=0;
      bs.aiShotStartX=bs.aiPosX; bs.aiShotStartY=bs.aiPosY;

      bs.aiShotTargetY=bs.hoopLY+(Math.random()-0.5)*14;
      bs.aiShotTimer=180+Math.floor(Math.random()*180);
      sndTennisHit();
    }

    if(bs.playerHasBall){
      const pd=Math.hypot(bs.playerX-bs.aiPosX,bs.playerY-bs.aiPosY);
      if(pd<36&&Math.random()<0.006){
        bs.playerHasBall=false; bs.aiHasBall=true;
        addFloat2(bs.aiPosX,bs.aiPosY-22,'CPU STEAL!','#e94560');
        sndTennisHit();
      }
    }
  } else {

    const defY=bs.playerY*0.4+(bs.courtY+bs.courtH/2)*0.6;

    bs.aiDir=bs.aiPosX<bs.playerX?3:2;

    if(bs.playerHasBall&&!bs.shooting){
      const pd=Math.hypot(bs.playerX-bs.aiPosX,bs.playerY-bs.aiPosY);
      if(pd<38&&Math.random()<0.006){

        addFloat2(bs.aiPosX,bs.aiPosY-22,'CPU STEAL! 🤚','#e94560');
        sndTennisHit();
        bs.aiBlockAnim=30;
      }
    }

    if(bs.shotMeterActive&&!bs.shooting){
      const pd=Math.hypot(bs.playerX-bs.aiPosX,bs.playerY-bs.aiPosY);
      const playerNearHoop=Math.hypot(bs.playerX-bs.hoopRX,bs.playerY-bs.hoopRY)<160;

        bs.shotMeterActive=false; bs.shotMeterPower=0;
        bs.playerHasBall=false;
        if(Math.random()<0.5){ bs.aiHasBall=true; }
        else { bs.playerHasBall=true; }
        addFloat2(bs.aiPosX,bs.aiPosY-30,'CPU BLOCK! 🙌','#fde047');
        sndTennisHit();
        bs.aiBlockAnim=35;
      }
    }
  }
}

function startPitch() {
  const bs = baseballState;
  bs.phase = 'pitch_incoming';
  bs.pitchX = bs.pitcherX;
  bs.pitchY = bs.pitcherY;
  const drift = (Math.random()-0.5)*40;
  bs.pitchTargetX = bs.homeX + drift;
  bs.pitchTargetY = bs.homeY;
  bs.pitchVX = (bs.pitchTargetX - bs.pitcherX) / 38;
  bs.pitchVY = (bs.pitchTargetY - bs.pitcherY) / 38;
  bs.pitchActive = true; bs.pitchT = 0;
  bs.swingQueued = false;
}

function doSwing() {
  const bs = baseballState;
  if(bs.phase !== 'pitch_incoming') return;
  bs.swingActive = true; bs.swingT = 0; bs.swingCD = 28;

  // Check contact: ball must be near home plate
  const distToPlate = Math.hypot(bs.pitchX - bs.homeX, bs.pitchY - bs.homeY);
  const contactWindow = 46 + bs.pitchT*0.5;
  if(distToPlate < contactWindow) {
    // Hit! Power based on timing (closer to plate = better)
    const timing = 1 - Math.min(1, distToPlate/46);
    const power = 0.5 + timing*0.5;
    bs.hitPower = power;
    bs.hitSuccess = true;
    bs.phase = 'batting';
    bs.ballX = bs.homeX; bs.ballY = bs.homeY;
    const hitAng = -Math.PI/2 + (Math.random()-0.5)*1.2;
    const spd = 4.5 + power*3.5;
    bs.ballVX = Math.cos(hitAng)*spd;
    bs.ballVY = Math.sin(hitAng)*spd - 1.5;
    bs.ballActive = true; bs.pitchActive = false;
    bs.runnerRunning = true; bs.runnerBase = 0;
    bs.runnerX = bs.homeX; bs.runnerY = bs.homeY;
    bs.fielderChasing = true; bs.fielderHasBall = false;
    const msg = power > 0.8 ? 'CRACK! GREAT HIT! 🔥' : 'HIT! ⚾';
    addFloat2(bs.homeX, bs.homeY-30, msg, power>0.8?'#fde047':'#4ade80');
    playTone(180,0.12,'sawtooth',0.2); setTimeout(()=>playTone(260,0.2,'square',0.18),40);
  } else {
    // Strike
    bs.hitSuccess = false;
    bs.pitchActive = false; bs.ballActive = false;
    bs.strikes++;
    if(bs.strikes >= 3) {
      bs.strikes = 0; bs.balls = 0; bs.outs++;
      addFloat2(bs.homeX, bs.homeY-30, 'STRIKEOUT! OUT '+(bs.outs)+'/3', '#ef4444');
      playTone(200,0.18,'sawtooth',0.15);
      if(bs.outs >= 3) endInning(); else { bs.phase='waiting'; bs.waitT=70; }
    } else {
      addFloat2(bs.homeX, bs.homeY-28, 'STRIKE '+bs.strikes+'!', '#f97316');
      playTone(220,0.12,'square',0.12);
      bs.phase='waiting'; bs.waitT=55;
    }
  }
}

function endInning() {
  const bs = baseballState;
  bs.outs=0; bs.balls=0; bs.strikes=0;
  // CPU "scores" between innings
  const cpuRuns = Math.floor(Math.random()*2);
  bs.aiScore += cpuRuns;
  if(cpuRuns>0) addFloat2(bs.cX+bs.cW/2, bs.cY+30, 'CPU SCORES '+cpuRuns+' RUN'+(cpuRuns>1?'S':'')+'!', '#ef4444');
  bs.innings++;
  if(bs.innings > bs.maxInnings || bs.playerScore>=9 || bs.aiScore>=9) {
    endGame(); return;
  }
  bs.flashMsg = 'INNING '+bs.innings+'!'; bs.flashT = 90;
  bs.phase='waiting'; bs.waitT=100;
}

function endGame() {
  const bs = baseballState;
  const won = bs.playerScore > bs.aiScore;
  bs.result = won ? 'won' : (bs.playerScore===bs.aiScore?'tie':'lost');
  const reward = won?50:bs.playerScore===bs.aiScore?15:8;
  coins += reward; updateHUD();
  addMsg(won?'⚾ Baseball Won! +'+reward+'🪙':'⚾ Baseball Lost. +'+reward+'🪙');
}

function updateBaseball() {
  const bs = baseballState;
  if(bs.result) return;
  if(bs.flashT>0) bs.flashT--;
  if(bs.swingCD>0) bs.swingCD--;

  // Animations
  bs.playerAnimT++; if(bs.playerAnimT%7===0) bs.playerFrame=(bs.playerFrame+1)%4;
  bs.cpuPitcherAnimT++; if(bs.cpuPitcherAnimT%10===0) bs.cpuPitcherFrame=(bs.cpuPitcherAnimT/10)%4;
  bs.fielderAnimT++; if(bs.fielderAnimT%8===0) bs.fielderFrame=(bs.fielderFrame+1)%4;

  if(bs.swingActive) {
    bs.swingT++;
    if(bs.swingT > 20) bs.swingActive=false;
  }

  if(bs.phase === 'waiting') {
    bs.waitT--;
    if(bs.waitT <= 0) startPitch();
    return;
  }

  if(bs.phase === 'pitch_incoming') {
    bs.pitchT++;
    bs.pitchX += bs.pitchVX;
    bs.pitchY += bs.pitchVY;
    // Check if pitch passed plate without swing
    if(bs.pitchY >= bs.homeY && bs.pitchActive) {
      bs.pitchActive = false;
      if(!bs.swingActive) {
        // Check strike zone (within ~30px of center)
        const inZone = Math.abs(bs.pitchX - bs.homeX) < 30;
        if(inZone) {
          bs.strikes++;
          if(bs.strikes >= 3) {
            bs.outs++;
            addFloat2(bs.homeX, bs.homeY-28, 'STRIKEOUT! OUT '+bs.outs+'/3','#ef4444');
            playTone(200,0.18,'sawtooth',0.15);
            if(bs.outs>=3) endInning(); else { bs.phase='waiting'; bs.waitT=65; }
          } else {
            addFloat2(bs.homeX, bs.homeY-26, 'STRIKE '+bs.strikes+'! (looking)', '#f97316');
            bs.phase='waiting'; bs.waitT=55;
          }
        } else {
          bs.balls++;
          if(bs.balls>=4) {
            addFloat2(bs.homeX, bs.homeY-26, 'WALK! Take your base!','#4ade80');
            bs.balls=0; bs.strikes=0;
            bs.runnerRunning=true; bs.runnerBase=0; bs.runnerX=bs.homeX; bs.runnerY=bs.homeY;
            bs.phase='running'; bs.fielderChasing=false;
          } else {
            addFloat2(bs.homeX, bs.homeY-26, 'BALL '+bs.balls+'!','#6ee7b7');
            bs.phase='waiting'; bs.waitT=55;
          }
        }
      }
    }
    // Auto-swing queued? (from SPACE key pressed ahead of pitch)
    if(bs.swingQueued && bs.pitchY > bs.homeY-50) {
      bs.swingQueued=false;
      doSwing();
    }
  }

  if(bs.phase === 'batting') {
    bs.ballVY += 0.15; // gravity
    bs.ballX += bs.ballVX;
    bs.ballY += bs.ballVY;
    // Bounce off field walls
    if(bs.ballX < bs.cX+10){ bs.ballVX=Math.abs(bs.ballVX)*0.6; bs.ballX=bs.cX+10; }
    if(bs.ballX > bs.cX+bs.cW-10){ bs.ballVX=-Math.abs(bs.ballVX)*0.6; bs.ballX=bs.cX+bs.cW-10; }
    if(bs.ballY < bs.cY+10){ bs.ballVY=Math.abs(bs.ballVY)*0.5; bs.ballY=bs.cY+10; }
    if(bs.ballY > bs.homeY){ bs.ballVY=-Math.abs(bs.ballVY)*0.45; bs.ballY=bs.homeY; if(Math.abs(bs.ballVY)<0.6){ bs.ballActive=false; } }

    // Fielder chases ball
    if(bs.fielderChasing && !bs.fielderHasBall) {
      const fdx=bs.ballX-bs.fielderX, fdy=bs.ballY-bs.fielderY;
      const fd=Math.hypot(fdx,fdy);
      if(fd>6) { bs.fielderX+=fdx/fd*1.8; bs.fielderY+=fdy/fd*1.8; bs.fielderFrame=(Math.floor(bs.fielderAnimT/6)%4); }
      if(fd<18 && !bs.ballActive) { bs.fielderHasBall=true; bs.fielderChasing=false; }
      if(fd<18 && bs.ballActive && Math.abs(bs.ballVX)<1 && Math.abs(bs.ballVY)<1) {
        bs.fielderHasBall=true; bs.fielderChasing=false; bs.ballActive=false;
      }
    }

    // Runner advances bases
    if(bs.runnerRunning) {
      const bases = [[bs.firstX,bs.firstY],[bs.secondX,bs.secondY],[bs.thirdX,bs.thirdY],[bs.homeX,bs.homeY]];
      const target = bases[Math.min(bs.runnerBase, 3)];
      const rdx=target[0]-bs.runnerX, rdy=target[1]-bs.runnerY;
      const rd=Math.hypot(rdx,rdy);
      if(rd>6){ bs.runnerX+=rdx/rd*2.2; bs.runnerY+=rdy/rd*2.2; }
      if(rd<10) {
        bs.runnerBase++;
        if(bs.runnerBase>=4) {
          // Scored a run!
          bs.runnerRunning=false; bs.playerScore++;
          const isHR = bs.hitPower>0.88;
          addFloat2(bs.homeX, bs.homeY-36, isHR?'HOME RUN! ⚾🎉 +1':'RUN SCORED! +1','#fde047');
          playTone(523,0.08,'square',0.14); setTimeout(()=>playTone(784,0.1,'square',0.14),80); setTimeout(()=>playTone(1047,0.14,'square',0.14),160);
          if(bs.playerScore>=9){ endGame(); return; }
          bs.balls=0; bs.strikes=0;
          bs.phase='waiting'; bs.waitT=80;
        }
      }
    }

    // Fielder catches/outs runner
    if(bs.fielderHasBall && bs.runnerRunning) {
      const bases = [[bs.firstX,bs.firstY],[bs.secondX,bs.secondY],[bs.thirdX,bs.thirdY],[bs.homeX,bs.homeY]];
      const targetBase = bases[Math.min(bs.runnerBase,3)];
      const fdx=targetBase[0]-bs.fielderX, fdy=targetBase[1]-bs.fielderY;
      const fd=Math.hypot(fdx,fdy);
      if(fd>5) { bs.fielderX+=fdx/fd*2.5; bs.fielderY+=fdy/fd*2.5; }
      // If fielder reaches base before runner
      if(fd<20) {
        const rdx2=Math.hypot(bs.runnerX-targetBase[0],bs.runnerY-targetBase[1]);
        if(rdx2>18) {
          bs.runnerRunning=false; bs.outs++;
          addFloat2(targetBase[0], targetBase[1]-22, 'RUNNER OUT! '+bs.outs+'/3','#ef4444');
          playTone(180,0.16,'sawtooth',0.14);
          if(bs.outs>=3) { endInning(); return; }
          bs.balls=0; bs.strikes=0; bs.phase='waiting'; bs.waitT=70;
        }
      }
    }
  }

  if(bs.phase==='running') {
    // Walk: runner moves to first
    const rdx=bs.firstX-bs.runnerX, rdy=bs.firstY-bs.runnerY;
    const rd=Math.hypot(rdx,rdy);
    if(rd>6){ bs.runnerX+=rdx/rd*1.6; bs.runnerY+=rdy/rd*1.6; }
    if(rd<10){ bs.runnerRunning=false; bs.balls=0; bs.strikes=0; bs.phase='waiting'; bs.waitT=65; }
  }
}

// ── SHOP INTERIOR ──
function drawClock() {
  const tod = gameTime / DAY_LEN;
  const totalMinutes = Math.floor(tod * 24 * 60);
  const hours = Math.floor(totalMinutes / 60) % 24;
  const minutes = totalMinutes % 60;
  const ampm = hours >= 12 ? 'PM' : 'AM';
  const h12 = hours % 12 === 0 ? 12 : hours % 12;
  const timeStr = String(h12).padStart(2,'0') + ':' + String(minutes).padStart(2,'0') + ' ' + ampm;

  const isNight = tod < 0.15 || tod > 0.82;
  const clockX = W - 150, clockY = H - 38;
  const clockW = 142, clockH = 28;

  R(clockX, clockY, clockW, clockH, 'rgba(0,0,0,0.8)');
  O(clockX, clockY, clockW, clockH, isNight ? '#6366f1' : '#fbbf24', 1.5);

  const icon = isNight ? '🌙' : tod > 0.55 ? '🌇' : '☀️';
  ctx.font = '13px serif';
  ctx.fillText(icon, ~~(clockX + 6), ~~(clockY + 19));

  T(timeStr, clockX + 26, clockY + 19,
    isNight ? '#a5b4fc' : '#fde68a', 7);

  const label = isNight ? 'NIGHT' : 'DAY ' + day;
  T(label, clockX + 90, clockY + 19,
    isNight ? '#818cf8' : '#fbbf24', 5);
}

      t += durS;
    }
    t = startT;
    for (const [freq, dur] of bass) {
      const durS = dur * BEAT;
      scheduleNote(freq * 0.5, t, durS * 0.75, 'triangle', bassVolMap[season]);
      if (season === 'autumn' && freq !== 0)
        scheduleNote(freq * 0.75, t, durS * 0.6, 'sine', bassVolMap[season] * 0.35);
      t += durS;
    }
    const barDur = 4 * BEAT;
    const numBars = Math.ceil(loopDur / barDur);
    for (let bar = 0; bar < numBars; bar++) {
      for (let slot = 0; slot < 8; slot++) {
        const hitT = startT + bar * barDur + slot * E8;
        if (hitT < startT + loopDur) scheduleDrum(drumBar[slot], hitT);
      }
    }
    const nextStart = startT + loopDur;
    bgMusic.loopTimer = setTimeout(() => {
      if (!bgMusic.stopped) scheduleLoop(nextStart);
    }, Math.max(0, (nextStart - ac.currentTime) * 1000 - 100));
  }

  scheduleLoop(ac.currentTime + 0.1);
}

function stopBgMusic() {
  if (!bgMusic) return;
  bgMusic.stopped = true;
  clearTimeout(bgMusic.loopTimer);
  try {
    bgMusic.masterGain.gain.setValueAtTime(bgMusic.masterGain.gain.value, bgMusic.ac.currentTime);
    bgMusic.masterGain.gain.linearRampToValueAtTime(0, bgMusic.ac.currentTime + 0.5);
  } catch(e) {}
  bgMusicPlaying = false;
  bgMusicSeason = null;
  bgMusic = null;
}

function updateTown(){
  let mx=0,my=0;
  if(keys['ArrowLeft']||keys['a']||keys['A']){mx=-player.speed;player.dir=2;}
  if(keys['ArrowRight']||keys['d']||keys['D']){mx=player.speed;player.dir=3;}
  if(keys['ArrowUp']||keys['w']||keys['W']){my=-player.speed;player.dir=1;}
  if(keys['ArrowDown']||keys['s']||keys['S']){my=player.speed;player.dir=0;}
  if(mx&&my){mx*=0.707;my*=0.707;}
  const nx=Math.max(10,Math.min(MAP_W-10,player.x+mx));
  const ny=Math.max(10,Math.min(MAP_H-10,player.y+my));
  let blocked=false;
  for(const b of BUILDINGS){
    if(b.type==='houseplot'||b.type==='myshop') continue;
    if(buildingCollides(b.x,b.y,b.w,b.h,nx,ny)){ blocked=true; break; }
  }
  if(!blocked){
    for(const t of TREES){
      if(Math.hypot(nx-t.x,ny-t.y)<12){ blocked=true; break; }
    }
  }
  if(!blocked){ player.x=nx; player.y=ny; }
  if(mx||my){ player.animT++; if(player.animT%7===0) player.frame=(player.frame+1)%4; }
  cam.x=Math.max(0,Math.min(MAP_W-W,player.x-W/2));
  cam.y=Math.max(0,Math.min(MAP_H-H,player.y-H/2));
  buildingCustomers.forEach(c=>{
    c.animT++;
    if(c.state==='waiting'){ c.waitT++; if(c.waitT>c.maxWait){ c.state='leaving'; c.wanderT=0; addFloat(c.x,c.y-20,'Left 😞','#e94560'); } }
   if(c.state==='happy'){ c.leaveT--; if(c.leaveT<=0){ c.state='wander'; c.wanderT=0; c.wanderDur=300+Math.floor(Math.random()*200); c.vx=(Math.random()-.5)*0.5; c.vy=(Math.random()-.5)*0.5; } }
    if(c.state==='wander'){
      c.wanderT++;
      c.x += c.vx; c.y += c.vy;
      // Bounce off map edges (stay on roads/open areas, not inside buildings)
      c.x = Math.max(20, Math.min(MAP_W-20, c.x));
      c.y = Math.max(20, Math.min(MAP_H-20, c.y));
      // Hard collision vs buildings
      for(const b of BUILDINGS){
        const pad=10;
        const bx1=b.x-pad, by1=b.y-pad, bx2=b.x+b.w+pad, by2=b.y+b.h+pad;
        if(c.x>bx1&&c.x<bx2&&c.y>by1&&c.y<by2){
          const overL=c.x-bx1, overR=bx2-c.x, overT=c.y-by1, overB=by2-c.y;
          const minO=Math.min(overL,overR,overT,overB);
          if(minO===overL){c.x=bx1;c.vx=-Math.abs(c.vx);}
          else if(minO===overR){c.x=bx2;c.vx=Math.abs(c.vx);}
          else if(minO===overT){c.y=by1;c.vy=-Math.abs(c.vy);}
          else{c.y=by2;c.vy=Math.abs(c.vy);}
        }
      }
      // Randomly change direction
      // Smoothly steer to a new direction every so often
      if(c.wanderT%100===0){ 
        const angle=Math.random()*Math.PI*2;
        c.vx+=(Math.cos(angle)*0.4-c.vx)*0.3;
        c.vy+=(Math.sin(angle)*0.4-c.vy)*0.3;
      }
      // Clamp to gentle walking speed
      const spd2=Math.hypot(c.vx,c.vy);
      if(spd2>0.55){c.vx=c.vx/spd2*0.55;c.vy=c.vy/spd2*0.55;}
      if(spd2<0.1){c.vx=(Math.random()-.5)*0.4;c.vy=(Math.random()-.5)*0.4;}
      c.animT++;
      c.dir=Math.abs(c.vx)>Math.abs(c.vy)?(c.vx>0?3:2):(c.vy>0?0:1);
      if(c.wanderT>=c.wanderDur){ c.state='leaving'; }
    }
    if(c.state==='leaving'){
      // Walk to nearest road path (y~250 is the horizontal road between rows)
      const roadY = 250;
      const tx = c.x, ty = roadY;
      const dx = tx-c.x, dy = ty-c.y;
      const dist = Math.hypot(dx,dy)||1;
      c.x += dx/dist*1.4; c.y += dy/dist*1.4;
      // Hard collision vs buildings while leaving
      for(const b of BUILDINGS){
        const pad=10;
        const bx1=b.x-pad, by1=b.y-pad, bx2=b.x+b.w+pad, by2=b.y+b.h+pad;
        if(c.x>bx1&&c.x<bx2&&c.y>by1&&c.y<by2){
          const overL=c.x-bx1, overR=bx2-c.x, overT=c.y-by1, overB=by2-c.y;
          const minO=Math.min(overL,overR,overT,overB);
          if(minO===overL){c.x=bx1;} else if(minO===overR){c.x=bx2;}
          else if(minO===overT){c.y=by1;} else{c.y=by2;}
        }
      }
      c.animT++;
      c.dir=c.vy<0?1:0;
      if(Math.abs(c.y-roadY)<8){
        const b2=BUILDINGS.find(bb=>bb.id===c.bid);
        Object.assign(c,makeCust(b2));
      }
    }
  });
  groundCoins.filter(c=>!c.collected).forEach(c=>{
    if(Math.hypot(player.x-c.x,player.y-c.y)<18){ c.collected=true; coins+=c.val; addFloat(c.x,c.y-10,'+'+c.val+'🪙','#ffd700'); sndCoin(); updateHUD(); }
  });
}

function openPanel(id){ if(id==='hpanel') buildHousePanel(); document.getElementById(id).style.display='flex'; }
function closePanel(id){ document.getElementById(id).style.display='none'; }
function buildHousePanel(){
  const grid=document.getElementById('bpgrid'); grid.innerHTML='';
  let done=0;
  HOUSE_ITEMS.forEach(item=>{
    if(item.done) done++;
    const canBuild=!item.done&&Object.entries(item.cost).every(([k,v])=>(inv[k]||0)>=v);
    const costStr=Object.entries(item.cost).map(([k,v])=>`${v}×${k}`).join(' ');
    const el=document.createElement('div');
    el.className='bp-card'+(item.done?' done':'');
    el.innerHTML=`<div class="big">${item.ico}</div><div class="ttl">${item.name}</div><div class="cst">${item.done?'✅ Built':costStr}</div>${!item.done?`<button ${canBuild?'':'disabled'} onclick="buildItem('${item.id}')">BUILD</button>`:''}`;
    grid.appendChild(el);
  });
  document.getElementById('hprog').textContent=`${done} / ${HOUSE_ITEMS.length} built`;
  document.getElementById('hpp').textContent=Math.round(done/HOUSE_ITEMS.length*100)+'%';
}
function buildItem(id){
  const item=HOUSE_ITEMS.find(h=>h.id===id);
  if(!item||item.done) return;
  if(!Object.entries(item.cost).every(([k,v])=>(inv[k]||0)>=v)){ addMsg('Not enough materials!');return; }
  Object.entries(item.cost).forEach(([k,v])=>inv[k]-=v);
  item.done=true;updateInvUI();buildHousePanel();addMsg('Built '+item.name+'! 🎉');sndBuild();
  for(let i=0;i<16;i++) particles.push({x:player.x+cam.x+(Math.random()-.5)*60,y:player.y+cam.y-20,vx:(Math.random()-.5)*6,vy:-Math.random()*6-2,color:['#ffd700','#22c55e','#e94560','#3b82f6'][Math.floor(Math.random()*4)],s:5+Math.random()*4,life:50,maxLife:50});
}

function draw(){
  ctx.clearRect(0,0,W,H);
  if(isMobile()){
    const exitBtn=document.getElementById('btn-exit');
    if(exitBtn){
      const needsExit=SCENE==='interior'||SCENE==='pcscreen'||SCENE==='shop'||SCENE==='tennis'||SCENE==='basketball'||SCENE==='baseball';
      exitBtn.style.display=needsExit?'flex':'none';
      const quitBtn=document.getElementById('btn-quit');
      if(quitBtn) quitBtn.style.display=SCENE==='town'?'flex':'none';
      // show/hide E, H, space buttons based on scene
      const eBtn=document.getElementById('btn-e');
      const hBtn=document.getElementById('btn-h');

      if(eBtn) eBtn.style.display=(SCENE==='town'||SCENE==='interior'||SCENE==='pcscreen'||SCENE==='shop')?'flex':'none';

      if(spBtn) spBtn.style.display=(SCENE==='tennis'||SCENE==='baseball')?'flex':'none';
      const shootBtn2=document.getElementById('btn-shoot');
      if(shootBtn2) shootBtn2.style.display=SCENE==='basketball'&&!bballState.result&&bballState.playerHasBall?'flex':'none';
      const sbBtn2=document.getElementById('btn-sb');
      if(sbBtn2) sbBtn2.style.display=SCENE==='basketball'&&!bballState.result?'flex':'none';
    }
  }
   if(SCENE==='town') { drawTown(); drawWeather(ctx); drawSeasons(); }
  else if(SCENE==='interior') drawInterior();
  else if(SCENE==='pcscreen') drawPCScreen();
  else if(SCENE==='shop') drawShopInterior();
  else if(SCENE==='tennis') drawTennisGame();
  else if(SCENE==='basketball') drawBballGame();
  else if(SCENE==='baseball') drawBaseballGame();
  else { R(0,0,W,H,'#0a0a12'); }
}

// ── MOBILE CONTROLS ──
function isMobile(){ return ('ontouchstart' in window) || navigator.maxTouchPoints > 0; }

function setupMobileControls(){
  if(!isMobile()) return;
  document.getElementById('dpad').style.display='block';
  const mb = document.getElementById('mbtns');
  mb.style.display='flex';
  mb.style.flexDirection='column';
  mb.style.gap='6px';
  mb.style.alignItems='flex-end';
  function holdKey(k){ return { start(){ keys[k]=true; }, stop(){ keys[k]=false; } }; }
  const map = {
    'btn-up':    holdKey('ArrowUp'),
    'btn-down':  holdKey('ArrowDown'),
    'btn-left':  holdKey('ArrowLeft'),
    'btn-right': holdKey('ArrowRight'),
  };
  Object.entries(map).forEach(([id,ctrl])=>{
    const el=document.getElementById(id);
    el.addEventListener('touchstart',e=>{e.preventDefault();ctrl.start();},{passive:false});
    el.addEventListener('touchend',  e=>{e.preventDefault();ctrl.stop(); },{passive:false});
    el.addEventListener('touchcancel',e=>{e.preventDefault();ctrl.stop();},{passive:false});
  });

  // E button
  document.getElementById('btn-e').addEventListener('touchstart',e=>{
    e.preventDefault();
    if(SCENE==='basketball'&&bballState.playerHasBall&&!bballState.shooting&&!bballState.result){
      bballState.shotMeterActive=true;bballState.shotMeterPower=0;bballState.shotMeterDir=1;
    } else { interact(); }
  },{passive:false});
  document.getElementById('btn-e').addEventListener('touchend',e=>{
    e.preventDefault();
    if(SCENE==='basketball'&&bballState.shotMeterActive&&!bballState.shooting&&bballState.playerHasBall&&!bballState.result){
      const bs=bballState;
      bs.shotMeterActive=false;bs.shooting=true;bs.shotArcT=0;
      bs.shotStartX=bs.playerX;bs.shotStartY=bs.playerY;
      bs.shotTargetX=bs.hoopRX;bs.shotTargetY=bs.hoopRY;
      bs.playerHasBall=false;sndTennisHit();
    }
  },{passive:false});

  // H button
  document.getElementById('btn-h').addEventListener('touchstart',e=>{
    e.preventDefault();
    const p=document.getElementById('hpanel');
    if(p.style.display==='flex') closePanel('hpanel'); else openPanel('hpanel');
  },{passive:false});

  // ● (space/swing) button
  document.getElementById('btn-space').addEventListener('touchstart',e=>{
    e.preventDefault();
    if(SCENE==='tennis'&&!tennisState.matchResult){ tennisState.swingQueued=true; }
    if(SCENE==='baseball'&&!baseballState.result){
      if(baseballState.phase==='pitch_incoming'&&!baseballState.swingActive&&baseballState.swingCD===0) doSwing();
      else baseballState.swingQueued=true;
    }
  },{passive:false});

  // SHOOT button (basketball only) — hold to charge, release to shoot
  const shootBtn = document.getElementById('btn-shoot');
  shootBtn.addEventListener('touchstart',e=>{
    e.preventDefault();
    if(SCENE!=='basketball'||bballState.result) return;
    const bs=bballState;
    if(bs.playerHasBall&&!bs.shooting){
      bs.shotMeterActive=true;
      bs.shotMeterPower=0;
      bs.shotMeterDir=1;
    }
  },{passive:false});
  shootBtn.addEventListener('touchend',e=>{
    e.preventDefault();
    if(SCENE!=='basketball'||bballState.result) return;
    const bs=bballState;
    if(bs.shotMeterActive&&!bs.shooting&&bs.playerHasBall){
      bs.shotMeterActive=false;
      bs.shooting=true;
      bs.shotArcT=0;
      bs.shotStartX=bs.playerX; bs.shotStartY=bs.playerY;
      bs.shotTargetX=bs.hoopRX; bs.shotTargetY=bs.hoopRY;
      bs.playerHasBall=false;
      sndTennisHit();
    }
  },{passive:false});
  shootBtn.addEventListener('touchcancel',e=>{
    e.preventDefault();
    bballState.shotMeterActive=false;
  },{passive:false});

  // STEAL/BLOCK button (basketball only)
  document.getElementById('btn-sb').addEventListener('touchstart',e=>{
    e.preventDefault();
    if(SCENE!=='basketball') return;
    const bs=bballState;
    if(bs.aiHasBall&&bs.stealCD===0){
      keys['q']=true; setTimeout(()=>{ keys['q']=false; },80);
    } else {
      keys['f']=true; setTimeout(()=>{ keys['f']=false; },80);
    }
  },{passive:false});

  // QUIT button (back to menu from town)
  document.getElementById('btn-quit').addEventListener('touchstart',e=>{
    e.preventDefault();
    if(SCENE==='town') goToMenu();
  },{passive:false});

  // EXIT button
  document.getElementById('btn-exit').addEventListener('touchstart',e=>{
    e.preventDefault();
    closePanel('hpanel');
    if(SCENE==='pcscreen'){ SCENE='interior';finishAct('pc');addMsg('Closed PC.'); }
    else if(SCENE==='interior'&&!hAct) exitHouse();
    else if(SCENE==='interior'&&hAct){ hAct=null;hProg=0;actAnimT=0;addMsg('Cancelled.'); }
    else if(SCENE==='shop') exitShop();
    else if(SCENE==='tennis') exitTennis();
    else if(SCENE==='basketball') exitBball();
    else if(SCENE==='baseball') exitBaseball();
  },{passive:false});

  // ESC via swipe-down on canvas
  canvas.addEventListener('touchstart',e=>{ touchStartY=e.touches[0].clientY; },{passive:true});
  canvas.addEventListener('touchend',e=>{
    const dy=e.changedTouches[0].clientY-touchStartY;
    if(dy>60){
      // swipe down = ESC
      closePanel('hpanel');
      if(SCENE==='pcscreen'){ SCENE='interior';finishAct('pc');addMsg('Closed PC.'); }
      else if(SCENE==='interior'&&!hAct) exitHouse();
      else if(SCENE==='interior'&&hAct){ hAct=null;hProg=0;actAnimT=0;addMsg('Cancelled.'); }
      else if(SCENE==='shop') exitShop();
      else if(SCENE==='tennis') exitTennis();
      else if(SCENE==='basketball') exitBball();
      else if(SCENE==='baseball') exitBaseball();
    }
  },{passive:true});
}
